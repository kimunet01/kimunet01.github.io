❗ 백준 1157번 문제 : `단어 공부`  
```
알파벳 대소문자로 된 단어가 주어지면, 이 단어에서 가장 많이 사용된 알파벳이 무엇인지 알아내는 프로그램을 작성하시오. 단, 대문자와 소문자를 구분하지 않는다.
```

```c++
#include <iostream>
using namespace std;

int main() {
	int arr[32]; //빈도 수 배열
	int count = 0;
	for (int i = 0; i < 32; ++i) {
		arr[i] = 0;
	}
	string s;
	cin >> s;
	for (int i = 0; i < s.length(); ++i) {
		arr[((int)s[i] - 64) % 32]++;
	}
	int max = arr[0]; //최대 빈도
	int max_n = 0;
	for (int i = 0; i < 32; ++i) { // i는 해당 아스키코드값
		if (arr[i] > max) {
			max = arr[i];
			max_n = i;
		}
	}
	for (int i = 0; i < 32; ++i) {
		if (arr[i] == arr[max_n]) {
			count++;
		}
	}
	if (count > 1) {
		cout << "?";
	}
	else {
		cout << (char)(max_n + 64);
	}
}
```
